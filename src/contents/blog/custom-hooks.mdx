## Introduction

Disini saya bagikan beberapa <i>custom hooks</i> yang dapat mempermudah proses development.

### 1. useFetchAPI

```jsx
import { useEffect, useState } from "react";

export const useFetchAPI = (url: string) => {
	const [data, setData] = useState();

	const [loading, setLoading] = useState(false);
	const [error, setError] = useState(null);

	useEffect(() => {
		setLoading(true);
		fetch(url)
			.then((res) => res.json())
			.then((data) => setData(data))
			.catch((err) => {
				setError(err);
				setLoading(false);
			})
			.finally(() => {
				setLoading(false);
			});
	}, [url]);

	const refetch = () => {
		setLoading(true);
		fetch(url)
			.then((res) => res.json())
			.then((data) => setData(data))
			.catch((err) => {
				setLoading(false);
				setError(err);
			})
			.finally(() => {
				setLoading(false);
			});
	};

	return { data, loading, error, refetch };
};
```
